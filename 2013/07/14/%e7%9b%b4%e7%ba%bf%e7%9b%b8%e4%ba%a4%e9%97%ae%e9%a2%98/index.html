直线相交问题

14 Jul 2013 bullet-and-wall checkio

最近在checkio上扫题的时候，碰到一个判断直线是否相交的问题，题目请猛击这里，这个题目以前上学的时候做到手软的好么，二话不说直接开写，先根据坐标上的两个点，算出斜率，然后得出直线表达式，最后根据两个直线表达式算出是否存在相交点，然后加上斜率为无穷大的情况等等，最后写出的代码如下。

[code lang="py"]
def checkio(data):
xw1, yw1 = data[0]
xw2, yw2 = data[1]
xa, ya = data[2]
xb, yb = data[3]

if(xw1 - xw2 == 0 and xb - xa == 0):
# vertical vertical
return False
elif(xw1 - xw2 == 0 and xb - xa != 0):
# ver hor
slope, d = cal_line(xa, ya, xb, yb)
y = xw1 * slope + d
dir = direction(data[2], data[3])
if(min(yw1, yw2) <= y <= max (yw1, yw2) and dir == 'left' and xw1 <= xa):
return True
elif((min(yw1, yw2) <= y <= max (yw1, yw2) and dir == 'right' and xw1 >= xa)):
return True
else:
return False
elif(xb - xa == 0 and xw1 - xw2 != 0):
# hor ver
slope, d = cal_line(xw1, yw1, xw2, yw2)
y = xa * slope + d
dir = direction(data[2], data[3])
if(min(xw1, xw2) <= xa <= max (xw1, xw2)and dir == 'down' and y <= ya):
return True
elif(min(xw1, xw2) <= xa <= max (xw1, xw2) and dir == 'up' and y >= ya):
return True
else:
return False
else:
# other
return common(data)

def common(data):
# common circumstance
xw1, yw1 = data[0]
xw2, yw2 = data[1]
xa, ya = data[2]
xb, yb = data[3]

slope_wall = float(yw1 - yw2) / float(xw1 - xw2)
slopex = float(yb - ya) / float(xb - xa)

d_wall = yw2 - slope_wall * xw2
d_x = yb - slopex * xb

dir = direction(data[2], data[3])

if(slope_wall == slopex and d_x != d_wall):
return False
elif(slope_wall == slopex and d_x == d_wall):
if(dir == "left" and min(xw1, xw2) <= xa):
return True
elif(dir == "right" and max(xw1, xw2) >= xa):
return True
else:
return False
else:
point_x = round(float(d_wall - d_x) / float(slopex - slope_wall), 1)
point_y = round(point_x * slope_wall + d_wall, 1)

if(dir == "right" and point_x >= xa and min(xw1, xw2) <= point_x <= max(xw1, xw2) and min(yw1, yw2) <= point_y <= max(yw1, yw2)):
return True
elif(dir == "left" and point_x <= xa and min(xw1, xw2) <= point_x <= max(xw1, xw2) and min(yw1, yw2) <= point_y <= max(yw1, yw2)):
return True
else:
return False

def direction(start, end):
xa, ya = start
xb, yb = end

if(xa == xb and xa > xb):
return 'down'
elif(xa == xb and xa < xb):
return 'up'
elif(xa != xb and xa > xb):
return 'left'
elif(xa != xb and xa < xb):
return 'right'

def cal_line(x1, y1, x2, y2):
slope = float(y1 - y2) / float(x1 - x2)
d = y1 - slope * x1
return [slope, d]

在调试了若干遍以后，程序是能跑起来了，但是这个程序，有这样几个问题。

1：计算过程中基本不可避免的会牵扯到浮点数的计算，而这种计算需要小心小心再小心。

2：浮点数中的误差等等问题在判断是否相交的时候是很麻烦的，我在程序里面使用了round让测试用例跑过了，但是并不意味着这样做就是合适的。

在查看checkio别人的slution的时候，发现此类题还有另外一种解法，即改变坐标系，用另外一种思路来考虑此问题，而且不用担心相交点的判定问题

如果直线AB和W1W2相交，那么AB的角度是是介于AW2和AW1的

[code lang="py"]
import math
def checkio(data):
xw1, yw1 = data[0]
xw2, yw2 = data[1]
xa, ya = data[2]
xb, yb = data[3]

w1 = math.atan2(yw1-ya, xw1-xa) % (2*math.pi)
w2 = math.atan2(yw2-ya, xw2-xa) % (2*math.pi)
b = math.atan2(yb-ya, xb-xa) % (2*math.pi)

l, r = min(w1, w2), max(w1, w2)
l, r = (r, (l + 2*math.pi)) if (l + 2*math.pi - r < r - l) else (l, r)

# point on the wall edge
if data[2] in data[:2]: return True

# point on the wall
if abs((r-l) - math.pi) < 1e-10: return True

# regular business
return (l <= b <= r) or (l <= b + 2*math.pi <= r)
if __name__ == '__main__':
assert checkio([[0,0], [0,2], [5,1], [3,1]]) == True, "First"
[/code]
